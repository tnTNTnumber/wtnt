Operatori di Appartenenza

Not In

Sintassi 

primo termine not in secondo termine

Esempio

Codice 
print ( 'a' not in 'ciao' )
Output 
False

Esempio

Codice 
print ( 'A' not in 'ciao' )
Output 
True

Esempio

Codice 
print ( 'q' not in 'ciao' )
Output 
True

Esempio

Codice 
print ( '' not in 'ciao' )
Output 
False

Esempio

Codice 
print ( '' not in '' )
Output 
False

Esempio

Codice 
X = 'Mostro'
print ( X not in X )
Output 
False

Esempio

Codice 
X = 'sacchetto'
Y = 'tt'
print ( Y not in X )
Output 
False

Esempio

Codice 
X = [ 1, 2, 3 ]
print ( 3 not in X )
Output 
False

Esempio

Codice 
X = [ 1, 2, 3 ]
print ( 5 not in X )
Output 
True

Esempio

Codice 
X = [ 'lupo', 'giraffa', 'camaleonte' ]
print ( 'azalea' not in X )
Output 
True

Esempio

Codice 
X = [ 'lupo', 'giraffa', 'camaleonte' ]
print ( 'giraffa' not in X )
Output 
False

Esempio

Codice 
X = [ 'emblema', 5, 'tetrade', True, '#' ]
print ( True not in X )
Output 
False

Esempio

Codice 
X = [ [ 1, 2 ], [ 3, 4 ] ]
print ( 3 not in X )
Output 
True

Esempio

Codice 
X = [ [ 1, 2 ], [ 3, 4 ] ]
print ( [ ] not in X )
Output 
True

Esempio

Codice 
X = [ [ 1, 2 ], [ 3, 4 ] ]
print ( [ 2 ] not in X )
Output 
True

Esempio

Codice 
X = [ [ 1, 2 ], [ 3, 4 ] ]
print ( [ 3, 4 ] not in X )
Output 
False

Esempio

Codice 
X = [ 1, 2, 3, 4 ]
print ( [ 2, 3 ] not in X )
Output 
True

Esempio

Codice 
X = [ 1, 2, 3, 4 ]
print ( '2' not in X )
Output 
True
